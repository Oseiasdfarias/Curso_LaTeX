    \section{Redefinindo comandos seno e tangente}
    
    Vamos redefinir as funções matemáticas do inglês para o português\\
    
   \textbf{ Equação seno}
    \begin{equation*}
    	\sin(2x)
    \end{equation*}

    \textbf{Equação Tangente}
    \begin{equation*}
    \tan(2x)
    \end{equation*}
    
    \textbf{Frações entre parênteses}
    \begin{equation*}
    	\left(\frac{2a}{3e}\right)'
    \end{equation*}
    
   \textbf{ Equação entre colchetes}
    \begin{equation*}
    	\{2a\}
    \end{equation*}
    
    \textbf{Porcentagem}
    \begin{equation*}
    100 \%
    \end{equation*}
    
    \textbf{Sub escrito }
	\begin{equation*}
	x_{12}
	\end{equation*}
